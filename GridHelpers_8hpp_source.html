43class SparseTableView
44{
45public:
47 using row_type = iterator_range_pod<int>;
48
54 SparseTableView(int* data, unsigned *offset, std::size_t size_arg)
55 : data_(data), offset_(offset), size_(size_arg)
56 {}
57
61 row_type operator[](std::size_t row) const
62 {
63 assert(row<=size());
64 return row_type{data_ + offset_[row],
65 data_ + offset_[row+1]};
66 }
67
70 std::size_t size() const
71 {
72 return size_;
73 }
74
76 std::size_t noEntries() const
77 {
78 return offset_[size_];
79 }
80
81private:
83 const int* data_;
87 const unsigned* offset_;
89 std::size_t size_;
90};
